        let timeLeft = 24 * 60 * 60; // 24 часа в секундах
        
        function updateTimer() {
            const hours = Math.floor(timeLeft / 3600);
            const minutes = Math.floor((timeLeft % 3600) / 60);
            const seconds = timeLeft % 60;
            
            document.getElementById('timer').textContent = 
                `Осталось времени: ${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}`;
            
            if (timeLeft > 0) {
                timeLeft--;
            } else {
                document.getElementById('timer').textContent = 'Скоро откроемся!';
            }
        }
        
        setInterval(updateTimer, 1000);
